Test for MyList module
=======================

import the class:
	>>> MyList =  __import__("1-my_list").MyList

check for empty list:
	>>> m = MyList()
	>>> m
	[]
	>>> m.print_sorted()
	[]

check for single element:
	>>> m = MyList()
	>>> m.append(1)
	>>> m
	[1]
        >>> m.print_sorted()
	[1]

check for sorted list:
	>>> m = MyList()
        >>> m.append(1)
	>>> m.append(2)
	>>> m.append(3)
	>>> m
	[1, 2, 3]
        >>> m.print_sorted()
        [1, 2, 3]

check for negative numbers:
        >>> m = MyList()
	>>> m.append(-10)
        >>> m.append(-2)
        >>> m.append(-13)
        >>> m
        [-10, -2, -13]
        >>> m.print_sorted()
        [-13, -10, -2]

check for float numbers:
        >>> m = MyList([1.5, 2.3, 3.09])
	>>> m
	[1.5, 2.3, 3.09]
        >>> m.print_sorted()
        [1.5, 2.3, 3.09]

check for strings:
	>>> m = MyList(["Best", "School", "ALX"])
	>>> m
	['Best', 'School', 'ALX']
        >>> m.print_sorted()
        ['ALX', 'Best', 'School']

check for string argument:
	>>> m = MyList("python")
	>>> m
	['p', 'y', 't', 'h', 'o', 'n']
	>>> m.print_sorted()
	['h', 'n', 'o', 'p', 't', 'y']
